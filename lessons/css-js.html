CSS Example:

<style>
    /* Select by element */
    p {
        color: blue;
        font-size: 16px;
    }
    
    /* Select by class */
    .highlight {
        background-color: yellow;
        padding: 10px;
    }
    
    /* Select by ID */
    #main-title {
        font-size: 2em;
        color: #333;
    }
    
    /* Hover effect */
    button:hover {
        background-color: #4CAF50;
        color: white;
    }
</style>
